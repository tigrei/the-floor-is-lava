ships_updated 

ship_id
	Type: string
	Description: Unique identifier for the ship
	Example: "DDG-51"
name
	Type: string
	Description: Official ship name
	Example: "USS Arleigh Burke"
cargo_capacity_tons
	Type: number
	Description: Usable cargo / payload capacity in metric tons
		Notes:
		Warships = mission supplies / usable storage
		Cargo ships = actual cargo tonnage
		Minimum: 0
weight_tons
	Type: number
	Description: Ship displacement (total operational weight) in metric tons
	Minimum: 0
max_speed_knots
	Type: number
	Description: Maximum sustainable speed in knots
	Minimum: 0
fuel_capacity_tons
	Type: number
	Description: Total fuel storage capacity in metric tons
	Minimum: 0
size
	Object containing ship dimensions in meters:

	x (number): length overall (LOA)
	y (number): beam (width)
